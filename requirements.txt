# Precision-Land Python dependencies
#
# Target: Python 3.9+ (3.12/3.13 on Pi is supported by current OpenCV wheels).
# OpenCV 4.10+ wheels are built for NumPy 2; pin both so pip does not mix an old
# cv2 (NumPy-1 ABI) with numpy 2.x (causes "_ARRAY_API not found").
#
# Raspberry Pi / Debian "externally managed" Python (PEP 668): you may need
# --break-system-packages when installing into the same interpreter that uses
# libcamera / Picamera2 (often no venv, because those stacks expect system Python).
# That flag is the supported escape hatch when you accept managing versions yourself.
#
#   python3 -m pip install --break-system-packages -r requirements.txt
#
# Optional: keep OpenCV/NumPy entirely from apt (pairs stay consistent; contrib may
# differ from pip). Example:
#   sudo apt install python3-numpy python3-opencv
# Then only pip-install packages not satisfied by apt, still with --break-system-packages
# if pip refuses.
#
# Before upgrading OpenCV from pip, remove stale wheels so the right cv2 loads:
#   python3 -m pip uninstall -y opencv-python opencv-contrib-python opencv-python-headless

# NumPy 2.x (matches OpenCV wheel ABI)
numpy>=2.0.0,<3

# OpenCV with contrib (ArUco, etc.). Do not also install opencv-python — same cv2 module.
opencv-contrib-python==4.13.0.92

# MAVLink / DroneKit stack (pymavlink >=2.4.49 provides pymavlink.mavftp_op for scripts/mavftp.py)
pymavlink>=2.4.49,<3
dronekit==2.9.2

# LED Control (Adafruit CircuitPython)
adafruit-circuitpython-neopixel
adafruit-blinka
# Raspberry Pi 5: PyPI package is Blinka-based (not adafruit-circuitpython-*).
# aarch64 matches Pi 4 64-bit too; Pi 4 users should `pip install rpi_ws281x` and skip this line if needed.
Adafruit-Blinka-Raspberry-Pi5-Neopixel>=1.0.0rc2; platform_machine == "aarch64"
rpi_ws281x; platform_machine != "aarch64"
firebase-admin>=6.0.0
